<<create a database table named my_table with 3 columns>>

CREATE TABLE my_table(
	myColumn INT PRIMARY KEY AUTO_INCREMENT,
	myColumn2 VARCHAR(20),
	myColumn3 VARCHAR(20),
	myColumn4 VARCHAR(20)
);



<<COLUMN CONSTRAINTS:>>

PRIMARY KEY		- Id || PRIMARY KEY = NOT NULL + UNIQUE
AUTO_INCREMENT		- automatically assigns a primary key / id to each newly added row
NOT NULL		- the column must always be filled up
UNIQUE 			- unique
DEFAULT 'Undecided' 	- set 'Undecided' as the default value when the column is not filled up



<<delete the database table, my_table>>

DROP TABLE my_table;



<<add a new column named myColumn4>>

ALTER TABLE my_table
ADD myColumn4 VARCHAR(20);



<<delete the column, myColumn4>>

ALTER TABLE my_table
DROP COLUMN myColumn4;



<<show column details of my_table

DESCRIBE my_table;

-----------------------------------RETRIEVE------------------------------------------------



<>	<--	not equal to



<<retrieve ALL columns of all rows from database table, my_table>>

SELECT *
FROM my_table;



<<retrieve ONLY the column, myColumn2, of all rows from database table, my_table>>

SELECT myColumn2
FROM my_table;



<<these two queries do the same thing>>

SELECT *
FROM my_table
WHERE myColumn4 = 'Programmer' OR myColumn4 = 'Invincible' OR myColumn4 = 'Incompetent';

	||

SELECT *
FROM my_table
WHERE myColumn4 IN('Programmer', 'Invincible', 'Incompetent');



<<retrieve all columns of all rows (IN ORDER by myColumn2) from database table,
my_table>>

SELECT *
FROM my_table
ORDER BY myColumn2;



<<retrieve all columns of all rows (IN ASCENDING ORDER by myColumn2) from database
table, my_table>>

SELECT *
FROM my_table
ORDER BY myColumn2 ASC;



<<retrieve all columns of all rows (IN DESCENDING ORDER by myColumn2) from database
table, my_table>>

SELECT *
FROM my_table
ORDER BY myColumn2 DESC;



<<retrieves only the first 2 results of the query in the database table, my_table>>

SELECT *
FROM my_table
LIMIT 2;



<<retrieves only the unique rows in the database table, my_table>>

SELECT DISTINCT *
FROM my_table;



<<retrieves the number of rows that is not NULL in myColumn4 in the database table,
my_table>>

SELECT COUNT(myColumn4)
FROM my_table;



<<retrieves the sum of the rows in myColumn5 of the database table, my_table>>

SELECT SUM(myColumn5)
FROM my_table;



<<retrieves the average value of the rows in myColumn5 of the database table, my_table>>

SELECT AVG(myColumn5)
FROM my_table;



<<gets the total points then divides that total according to the team>>

SELECT SUM(points), team
FROM RECORD
GROUP BY team;
